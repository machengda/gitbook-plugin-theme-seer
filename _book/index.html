Gitbook theme for BingoPaaS Docs

Usage

Add the theme to your book's configuration book.json or book.js:

{
    "plugins": [
        "theme-seer"
    ],
    "variables": {
        "themeSeer":{
            "nav":[
                {
                    "url":"http://...",
                    "target":"_blank",
                    "name": "简易教程"
                }
            ]
        },
    },
    "pluginsConfig": {
        "theme-seer":{
            "search-placeholder":"输入关键字搜索", //搜索框提示信息
            "logo":"./logo.png", //logo地址
            "favicon": "./favicon.ico" //ico地址
        }
    }
},

Install by command:

gitbook install

Recommand plugins

plugins: [
    '-sharing',
    "-fontsettings",
    'back-to-top-button',
    "copy-code-button",
    "cuav-chapters",
    "heading-anchors",
    "theme-seer"
    //...
]

Enjoy!
